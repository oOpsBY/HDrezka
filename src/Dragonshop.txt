		if (this.m_LocalUserID == -999L)
		{
			this.ApiLog(EndGamesAPI.LogLevel.WARNING, "Anonymous user");
			this.m_LocalUser = this.GetUserInfo(this.m_LocalUserID);
			this.m_LocalUser.AccountIsLoaded = true;
			this.m_LocalUser.m_UserInfo.Handle = "Dragonlord";
		}
		else if (this.m_LocalUserID > 0L)
		{
			this.m_LocalUser.m_UserToken = this.m_LocalUserToken;
		}
		if (!string.IsNullOrEmpty(EndGamesAPI.m_Url))
		{
			try
			{
				Caching.maximumAvailableDiskSpace = 1073741824L;
				if (Application.webSecurityEnabled && this.m_RequestType != EndGamesAPIWWW.Type.WWW)
				{
					string text = NetworkUtility.ResolveAddress(new Uri(EndGamesAPI.m_Url).Host).ToString();
					int num = 5;
					bool flag = true;
					for (int i = 0; i < num; i++)
					{
						if (Security.PrefetchSocketPolicy(text, 843))
						{
							flag = false;
							break;
						}
					}
					if (flag)
					{
						this.ApiLog(EndGamesAPI.LogLevel.ERROR, "Socket Policy Prefetch Failure: " + text);
					}
					else
					{
						this.ApiLog(EndGamesAPI.LogLevel.NOTIFICATION, "Socket Policy Prefetch Success: " + text);
					}
				}
			}
			catch (Exception ex)
			{
				this.ApiLog(EndGamesAPI.LogLevel.ERROR, "URL Init Failure: " + ex.ToString());
			}
		}
		this.Preload();
		if (this.m_LocalUserID == -999L)
		{
			this.LoadFinished(this.m_LocalUserID, EndGamesAPI.LoadType.USER_INFO, true, string.Empty);
		}
		if (PlatformExtraInfo.IsTestMode() && this.m_LocalUserID > 0L)
		{
			this.LoadFriends(this.m_LocalUserID, this.m_LocalUserToken);
		}
		if (this.m_SaveUserOptionsRemotely || this.m_SaveUserOptionsLocally)

			{
				this.LoadAccount(this.m_LocalUserID, new EndGamesAPIModels.LoadAccountFlags(true, this.m_LoadForgeLoadouts));
				return;
			}
		}
		else
		{

			{
				this.m_Preloading.Add(EndGamesAPI.LoadType.GAME_INFO);
			}
			if (this.m_OptionsLoading == EndGamesAPI.LoadStyle.PRELOAD)
			{
				this.m_Preloading.Add(EndGamesAPI.LoadType.OPTIONS);
				this.m_Preloading.Add(EndGamesAPI.LoadType.DRAGON_SHOP_DETAILS);
			}
			if (this.m_UserLoading == EndGamesAPI.LoadStyle.PRELOAD)
			{
				this.m_Preloading.Add(EndGamesAPI.LoadType.USER_INFO);
			}
			if (this.m_GameInfoLoading != EndGamesAPI.LoadStyle.NONE)
			{
				this.LoadGameInfo();
			}
			if (this.m_OptionsLoading != EndGamesAPI.LoadStyle.NONE)
			{
				this.LoadOptions();
				this.LoadAllDragonShopDetails();
			}
			if (this.m_UserLoading != EndGamesAPI.LoadStyle.NONE && this.m_LocalUserID > 0L)
			{
				this.LoadAccount(this.m_LocalUserID, new EndGamesAPIModels.LoadAccountFlags(true, this.m_LoadForgeLoadouts));
			}
		}
	}

		this.ApiLog(EndGamesAPI.LogLevel.NOTIFICATION, string.Concat(new object[]
		{
			"API: LoadFinished: ",
			type.ToString(),
			", Success: ",
			success,
			", Msg: ",
			msg
		}));
		EndGamesAPI.UserInfo userInfo = this.GetUserInfo(EndAccountID);
		userInfo.m_IsLoading.Remove(type);
		if (!userInfo.m_HasLoaded.Contains(type))
		{
			userInfo.m_HasLoaded.Add(type);
		}
		switch (type)
		{
		case EndGamesAPI.LoadType.GAME_INFO:
			if (success)
			{
				Messenger.BroadcastToAllListeners("EndAPI:LoadGameInfoComplete");
			}
			else
			{
				Messenger<string>.BroadcastToAllListeners("EndAPI:LoadGameInfoError", ref msg);
			}
			break;
		case EndGamesAPI.LoadType.OPTIONS:
			if (success)
			{
				Messenger.BroadcastToAllListeners("EndAPI:LoadOptionsComplete");
			}
			else
			{
				Messenger<string>.BroadcastToAllListeners("EndAPI:LoadOptionsError", ref msg);
			}
			break;
		case EndGamesAPI.LoadType.USER_INFO:
			if (success)
			{
				Messenger<long>.BroadcastToAllListeners("EndAPI:LoadUserInfoCompleteForUser", ref EndAccountID);
				if (EndAccountID == this.m_LocalUserID || EndAccountID <= 0L)
				{
					Messenger.BroadcastToAllListeners("EndAPI:LoadUserInfoComplete");
					LocalizationManager.SetCurrency(this.m_LocalUser.m_UserInfo.CurrencyInfo);
				}
				if (EndAccountID == this.m_LocalUserID && this.m_OptionsLoading != EndGamesAPI.LoadStyle.NONE && this.m_Options == null)
				{
					this.LoadOptions();
					this.LoadAllDragonShopDetails();
				}
			}
			else
			{
				Messenger<long, string>.BroadcastToAllListeners("EndAPI:LoadUserInfoError", ref EndAccountID, ref msg);
			}
			break;
		case EndGamesAPI.LoadType.USER_INFO_REFRESH:
		case EndGamesAPI.LoadType.SAVE_USER_INFO:
			if (!userInfo.AccountIsLoaded)
			{
				this.ApiLog(EndGamesAPI.LogLevel.WARNING, "Got a " + type + " message before account was loaded!");
			}
			else if (success)
			{
				Messenger<long>.BroadcastToAllListeners("EndAPI:UserInfoRefreshedForUser", ref EndAccountID);
				if (EndAccountID == this.m_LocalUserID || EndAccountID <= 0L)
				{
					Messenger.BroadcastToAllListeners("EndAPI:UserInfoRefreshed");
				}
			}
			else
			{
				Messenger<long, string>.BroadcastToAllListeners("EndAPI:LoadUserInfoError", ref EndAccountID, ref msg);
			}
			break;
		case EndGamesAPI.LoadType.USER_OPTIONS:
			if (success)
			{
				Messenger<long>.BroadcastToAllListeners("EndAPI:LoadUserOptionsCompleteForUser", ref EndAccountID);
				if (EndAccountID == this.m_LocalUserID || EndAccountID <= 0L)
				{
					Messenger.BroadcastToAllListeners("EndAPI:LoadUserOptionsComplete");
				}
			}
			else
			{
				Messenger<long, string>.BroadcastToAllListeners("EndAPI:LoadUserOptionsError", ref EndAccountID, ref msg);
			}
			break;
		case EndGamesAPI.LoadType.SAVE_USER_OPTIONS:
			if (success)
			{
				Messenger<long>.BroadcastToAllListeners("EndAPI:UserOptionsSaved", ref EndAccountID);
			}
			else
			{
				Messenger<long, string>.BroadcastToAllListeners("EndAPI:SaveUserOptionsError", ref EndAccountID, ref msg);
			}
			break;
		case EndGamesAPI.LoadType.FRIENDS:
			if (success)
			{
				Messenger<long>.BroadcastToAllListeners("EndAPI:LoadFriendsCompleteForUser", ref EndAccountID);
				if (EndAccountID == this.m_LocalUserID || EndAccountID <= 0L)
				{
					Messenger.BroadcastToAllListeners("EndAPI:LoadFriendsComplete");
				}
			}
			else
			{
				Messenger<string>.BroadcastToAllListeners("EndAPI:LoadFriendsError", ref msg);
			}
			break;
		}
		if (!success)
		{
			this.ApiLog((type != EndGamesAPI.LoadType.FRIENDS) ? EndGamesAPI.LogLevel.ERROR : EndGamesAPI.LogLevel.WARNING, string.Format("API: LoadFailed [{0}] {1}", type, msg));
			Messenger<EndGamesAPI.LoadType, string>.BroadcastToAllListeners("EndAPI:LoadError", ref type, ref msg);
			if (this.m_Preloading.Contains(type))
			{
				this.m_PreloadError.Add(type, msg);
			}
		}
		if (this.m_Preloading.Count > 0)
		{
			this.m_Preloading.Remove(type);
			if (this.m_Preloading.Count == 0)
			{
				if (this.m_PreloadError.Count > 0)
				{
					this.ApiLog(EndGamesAPI.LogLevel.WARNING, "API: PreloadFailed");
					Messenger<Dictionary<EndGamesAPI.LoadType, string>>.BroadcastToAllListeners("EndAPI:PreloadFailed", ref this.m_PreloadError);
					return;
				}
				this.ApiLog(EndGamesAPI.LogLevel.NOTIFICATION, "API: PreloadComplete");
				Messenger.BroadcastToAllListeners("EndAPI:PreloadComplete");
			}
		}
	}
